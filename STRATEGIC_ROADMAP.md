### Component Architecture (No Hardcoding)
```
PAGES FETCH DATA DYNAMICALLY:

/servicii/page.tsx:
  const services = await db.query('services')
  → render <ServiceGrid services={services} />

/servicii/[slug]/page.tsx:
  const service = await db.query('services', { slug })
  const caseStudies = await db.query('case_studies', { service_id })
  → render <ServicePage service={service} caseStudies={caseStudies} />

/portofoliu/page.tsx:
  const caseStudies = await db.query('case_studies')
  → render <CaseStudyGrid caseStudies={caseStudies} />

/portofoliu/[slug]/page.tsx:
  const caseStudy = await db.query('case_studies', { slug })
  const related = await db.query('case_studies', { category: caseStudy.category })
  → render <CaseStudyPage caseStudy={caseStudy} />

/blog/page.tsx:
  const articles = await db.query('blog_articles', { limit: 10, offset })
  → render <BlogGrid articles={articles} />

/blog/[slug]/page.tsx:
  const article = await db.query('blog_articles', { slug })
  const related = await db.query('blog_articles', { tags: article.tags })
  → render <ArticlePage article={article} />

/pricing/page.tsx:
  const tiers = await db.query('pricing_tiers')
  const faqs = await db.query('faq', { category: 'pricing' })
  → render <PricingPage tiers={tiers} faqs={faqs} />
```

## 7️⃣ CODE QUALITY STANDARDS (CLEAN & RAPID)

### TypeScript First
```typescript
// ✅ GOOD
type Service = {
  id: string;
  slug: string;
  title: string;
  description: string;
  features: string[];
  pricing_start: number;
};

// ❌ BAD
const service = { name: "service", data: {...} };
```

### Component Pattern (Reusable)
```typescript
// ✅ GOOD - Reusable, testable
interface CardProps {
  title: string;
  description: string;
  image: string;
  cta: { text: string; href: string };
}

export function Card({ title, description, image, cta }: CardProps) {
  return (
    <div className="rounded-lg border bg-white p-6">
      <h2>{title}</h2>
      <p>{description}</p>
      <Link href={cta.href}>{cta.text}</Link>
    </div>
  );
}

// ❌ BAD - Hardcoded, not reusable
export function ServiceCard1() {
  return (
    <div>
      <h2>Web Development</h2>
      <p>Some hardcoded description</p>
      <Link href="/servicii/web-development">Learn more</Link>
    </div>
  );
}
```

### API Pattern (Error Handling)
```typescript
// ✅ GOOD
export async function POST(req: Request) {
  try {
    const body = await req.json();
    const validated = contactFormSchema.parse(body);
    
    // Save to DB
    const lead = await db.leads.create(validated);
    
    // Send email
    await resend.emails.send({
      from: "hello@programatorpersonal.ro",
      to: validated.email,
      template: "contact-confirmation",
    });
    
    return NextResponse.json({ success: true, id: lead.id });
  } catch (error) {
    console.error("Contact form error:", error);
    return NextResponse.json(
      { error: "Failed to submit form" },
      { status: 500 }
    );
  }
}

// ❌ BAD
export async function POST(req: Request) {
  const body = await req.json();
  // No validation
  // No error handling
  // No email sent
  return NextResponse.json({ ok: true });
}
```
